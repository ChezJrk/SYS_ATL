// memcpy(
//     N : size,
//     A : f32[N] @DRAM,
//     B : f32[N] @DRAM
// )
void memcpy( void *exo_restrict ctxt, int_fast32_t N, float*exo_restrict A, const float*exo_restrict B ) {
for (int_fast32_t i = 0; i < N; i++) {
  A[i] = B[i];
}
}

// memcpy_ab(
//     N : size,
//     A : [f32][N] @DRAM,
//     B : [f32][N] @DRAM
// )
void memcpy_ab( void *exo_restrict ctxt, int_fast32_t N, struct exo_win_1f32 A, struct exo_win_1f32c B ) {
for (int_fast32_t i = 0; i < N; i++) {
  A.data[i * A.strides[0]] = B.data[i * B.strides[0]];
}
}

// memcpy_b(
//     N : size,
//     A : f32[N] @DRAM,
//     B : [f32][N] @DRAM
// )
void memcpy_b( void *exo_restrict ctxt, int_fast32_t N, float*exo_restrict A, struct exo_win_1f32c B ) {
for (int_fast32_t i = 0; i < N; i++) {
  A[i] = B.data[i * B.strides[0]];
}
}
