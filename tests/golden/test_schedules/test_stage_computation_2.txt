def axpy(n: size, y: f32[n] @ DRAM, x: f32[n] @ DRAM, alpha: f32 @ DRAM):
    for i in seq(0, n):
        reg_1: f32
        reg_1 = x[i]
        reg_2: f32
        reg_2 = alpha
        reg_3: f32
        reg_3 = reg_1 * reg_2
        reg_4: f32 @ DRAM
        reg_4 = y[i]
        reg_4 += reg_3
        y[i] = reg_4
