def alloc_nest(n: size, m: size, x: R[n_14, m_15], y: R[n_14, m_15] @ DRAM,
               res: R[n_14, m_15] @ DRAM):
    # @instr TEST
    for i in par(0, n):
        rloc: R[m_15] @ DRAM
        xloc: R[m_15] @ DRAM
        yloc: R[m_15] @ DRAM
        for j in par(0, m):
            xloc[j] = x[i, j]
        for j in par(0, m):
            yloc[j] = y[i, j]
        for j in par(0, m):
            rloc[j] = xloc[j] + yloc[j]
        for j in par(0, m):
            res[i, j] = rloc[j]
