struct systl_win_1f32{
    float *data;
    int_fast32_t strides[1];
};
typedef struct c_code_str_Context { 

} c_code_str_Context;

static int _floor_div(int num, int quot) {
  int off = (num>=0)? 0 : quot-1;
  return (num-off)/quot;
}

static int8_t _clamp_32to8(int32_t x) {
  return (x < -128)? -128 : ((x > 127)? 127 : x);
}

#include <stdio.h>
#include <stdlib.h>


// dot(
//     m : size,
//     x : [f32][m]  @DRAM,
//     y : [f32][m]  @DRAM,
//     r : f32  @DRAM
// )
void dot( c_code_str_Context *ctxt, int_fast32_t m, struct systl_win_1f32 x, struct systl_win_1f32 y, float* r ) {
*r = 0.0;
for (int i = 0; i < m; i++) {
  *r += x.data[(i) * (x.strides[0])] * y.data[(i) * (y.strides[0])];
}
}

// proj(
//     n : size,
//     m : size,
//     x : f32[n,m]  @DRAM,
//     y : f32[m,n]  @DRAM
// )
void proj( c_code_str_Context *ctxt, int_fast32_t n, int_fast32_t m, float* x, float* y ) {
SYS_ATL_ASSUME(n > 4);
SYS_ATL_ASSUME(m > 4);
float xy;
float y2;
dot(ctxt,m,(struct systl_win_1f32){ (float*)&x[(1) * (m) + (0) * (1)], { 1 } },(struct systl_win_1f32){ (float*)&y[(0) * (n) + (2) * (1)], { n } },&xy);
dot(ctxt,m,(struct systl_win_1f32){ (float*)&y[(0) * (n) + (3) * (1)], { n } },(struct systl_win_1f32){ (float*)&y[(0) * (n) + (3) * (1)], { n } },&y2);
}
